create table public.alumnos (
  id uuid not null,
  nombre text null,
  username text null,
  email text null,
  telefono text null,
  departamento text null,
  ciudad text null,
  bio text null,
  instagram text null,
  linkedin text null,
  facebook text null,
  github text null,
  idiomas jsonb null,
  habilidades text null,
  doc_tipo text null,
  doc_numero text null,
  universidad_id uuid null,
  grupo_id uuid null,
  area text null,
  foto_perfil text null,
  foto_doc text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint alumnos_pkey primary key (id),
  constraint alumnos_username_key unique (username),
  constraint alumnos_grupo_id_fkey foreign KEY (grupo_id) references grupos (id) on delete set null,
  constraint alumnos_id_fkey foreign KEY (id) references auth.users (id),
  constraint alumnos_universidad_id_fkey foreign KEY (universidad_id) references universidades (id) on delete set null
) TABLESPACE pg_default;

create table public.estudiantes (
  id uuid not null default gen_random_uuid (),
  universidad_id uuid not null,
  grupo_id uuid not null,
  nombre_completo text not null,
  email text null,
  matricula text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  telefono text null,
  direccion text null,
  constraint estudiantes_pkey primary key (id),
  constraint estudiantes_grupo_id_fkey foreign KEY (grupo_id) references grupos (id) on delete CASCADE,
  constraint estudiantes_universidad_id_fkey foreign KEY (universidad_id) references universidades (id) on delete CASCADE

create table public.evaluaciones_alumnos (
  id uuid not null default gen_random_uuid (),
  empresa_id uuid not null,
  estudiante_id uuid not null,
  grupo_id uuid not null,
  puntualidad integer null,
  disciplina integer null,
  responsabilidad integer null,
  respeto integer null,
  desempeno integer null,
  comentario text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint evaluaciones_alumnos_pkey primary key (id),
  constraint evaluaciones_alumnos_empresa_id_fkey foreign KEY (empresa_id) references empresas (id) on delete CASCADE,
  constraint evaluaciones_alumnos_estudiante_id_fkey foreign KEY (estudiante_id) references estudiantes (id) on delete CASCADE,
  constraint evaluaciones_alumnos_grupo_id_fkey foreign KEY (grupo_id) references grupos (id) on delete CASCADE,
  constraint evaluaciones_alumnos_desempeno_check check (
    (
      (desempeno >= 1)
      and (desempeno <= 5)
    )
  ),
  constraint evaluaciones_alumnos_responsabilidad_check check (
    (
      (responsabilidad >= 1)
      and (responsabilidad <= 5)
    )
  ),
  constraint evaluaciones_alumnos_disciplina_check check (
    (
      (disciplina >= 1)
      and (disciplina <= 5)
    )
  ),
  constraint evaluaciones_alumnos_puntualidad_check check (
    (
      (puntualidad >= 1)
      and (puntualidad <= 5)
    )
  ),
  constraint evaluaciones_alumnos_respeto_check check (
    (
      (respeto >= 1)
      and (respeto <= 5)
    )
  )
) TABLESPACE pg_default;

create table public.grupos (
  id uuid not null default gen_random_uuid (),
  universidad_id uuid not null,
  nombre_grupo text not null,
  especialidad text null,
  fecha_creacion timestamp with time zone not null default timezone ('utc'::text, now()),
  horas_requeridas integer null,
  codigo_acceso character varying(8) null,
  constraint grupos_pkey primary key (id),
  constraint grupos_codigo_acceso_key unique (codigo_acceso),
  constraint grupos_universidad_id_fkey foreign KEY (universidad_id) references universidades (id) on delete CASCADE,
  constraint grupos_horas_requeridas_check check (
    (
      (horas_requeridas >= 1)
      and (horas_requeridas <= 500)
    )
  )
) TABLESPACE pg_default;

create table public.solicitudes_horas (
  id uuid not null default gen_random_uuid (),
  universidad_id uuid not null,
  empresa_id uuid not null,
  grupo_id uuid not null,
  estado text null default 'pendiente'::text,
  motivo_rechazo text null,
  decision_universidad text null default 'pendiente'::text,
  motivo_rechazo_universidad text null,
  hora_inicio time without time zone null,
  hora_fin time without time zone null,
  dias_asistencia text null,
  fecha_inicio date null,
  fecha_fin date null,
  direccion_maps text null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint solicitudes_horas_pkey primary key (id),
  constraint solicitudes_horas_empresa_id_fkey foreign KEY (empresa_id) references empresas (id) on delete CASCADE,
  constraint solicitudes_horas_grupo_id_fkey foreign KEY (grupo_id) references grupos (id) on delete CASCADE,
  constraint solicitudes_horas_universidad_id_fkey foreign KEY (universidad_id) references universidades (id) on delete CASCADE,
  constraint solicitudes_horas_decision_universidad_check check (
    (
      decision_universidad = any (
        array[
          'pendiente'::text,
          'aprobada'::text,
          'rechazada'::text
        ]
      )
    )
  ),
  constraint solicitudes_horas_estado_check check (
    (
      estado = any (
        array[
          'pendiente'::text,
          'en_revision'::text,
          'aprobada'::text,
          'cerrada'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;
